        class EcosortDatabase {
            constructor() {
                this.dbName = 'EcosortDB';
                this.dbVersion = 1;
                this.db = null;
            }

            async init() {
                return new Promise((resolve, reject) => {
                    const request = indexedDB.open(this.dbName, this.dbVersion);
                    request.onerror = (event) => { console.error('Database error:', event.target.error);
                        reject('Error opening database'); };
                    request.onsuccess = (event) => { this.db = event.target.result;
                        console.log('✅ Database connected');
                        resolve(this.db); };
                    request.onupgradeneeded = (event) => {
                        const db = event.target.result;
                        if (!db.objectStoreNames.contains('users')) {
                            const userStore = db.createObjectStore('users', { keyPath: 'email' });
                            userStore.createIndex('name', 'name', { unique: false });
                            userStore.createIndex('address', 'address', { unique: false });
                            userStore.createIndex('contactNumber', 'contactNumber', { unique: false });
                            userStore.createIndex('createdAt', 'createdAt', { unique: false });
                        }
                        if (!db.objectStoreNames.contains('pickups')) {
                            const pickupStore = db.createObjectStore('pickups', { keyPath: 'id',
                                autoIncrement: true });
                            pickupStore.createIndex('userEmail', 'userEmail', { unique: false });
                            pickupStore.createIndex('status', 'status', { unique: false });
                        }
                    };
                });
            }

            async addUser(userData) {
                return new Promise((resolve, reject) => {
                    const transaction = this.db.transaction(['users'], 'readwrite');
                    const store = transaction.objectStore('users');
                    const getRequest = store.get(userData.email);
                    getRequest.onsuccess = (event) => {
                        if (event.target.result) { reject('Email already registered'); return; }
                        userData.createdAt = new Date().toISOString();
                        userData.ecoPoints = userData.ecoPoints || 0;
                        userData.totalPickups = userData.totalPickups || 0;
                        userData.history = userData.history || [];
                        const addRequest = store.add(userData);
                        addRequest.onsuccess = () => { console.log('✅ User added:', userData.email);
                            resolve(userData); };
                        addRequest.onerror = () => { reject('Error adding user'); };
                    };
                });
            }

            async getUser(email) {
                return new Promise((resolve, reject) => {
                    const transaction = this.db.transaction(['users'], 'readonly');
                    const store = transaction.objectStore('users');
                    const request = store.get(email);
                    request.onsuccess = (event) => { resolve(event.target.result || null); };
                    request.onerror = () => { reject('Error fetching user'); };
                });
            }

            async getAllUsers() {
                return new Promise((resolve, reject) => {
                    const transaction = this.db.transaction(['users'], 'readonly');
                    const store = transaction.objectStore('users');
                    const request = store.getAll();
                    request.onsuccess = (event) => { resolve(event.target.result || []); };
                    request.onerror = () => { reject('Error fetching users'); };
                });
            }

            async updateUser(userData) {
                return new Promise((resolve, reject) => {
                    const transaction = this.db.transaction(['users'], 'readwrite');
                    const store = transaction.objectStore('users');
                    const request = store.put(userData);
                    request.onsuccess = () => { console.log('✅ User updated:', userData.email);
                        resolve(userData); };
                    request.onerror = () => { reject('Error updating user'); };
                });
            }

            async searchUsers(searchTerm) {
                const users = await this.getAllUsers();
                const term = searchTerm.toLowerCase();
                return users.filter(user =>
                    (user.name && user.name.toLowerCase().includes(term)) ||
                    user.email.toLowerCase().includes(term) ||
                    (user.address && user.address.toLowerCase().includes(term)) ||
                    (user.contactNumber && user.contactNumber.includes(term))
                );
            }

            async addPickup(pickupData) {
                return new Promise((resolve, reject) => {
                    const transaction = this.db.transaction(['pickups'], 'readwrite');
                    const store = transaction.objectStore('pickups');
                    pickupData.createdAt = new Date().toISOString();
                    const request = store.add(pickupData);
                    request.onsuccess = (event) => { pickupData.id = event.target.result;
                        resolve(pickupData); };
                    request.onerror = () => { reject('Error adding pickup'); };
                });
            }

            async getTotalUsers() {
                return new Promise((resolve, reject) => {
                    const transaction = this.db.transaction(['users'], 'readonly');
                    const store = transaction.objectStore('users');
                    const request = store.count();
                    request.onsuccess = (event) => { resolve(event.target.result); };
                    request.onerror = () => { reject('Error counting users'); };
                });
            }

            async getAllPickups() {
                return new Promise((resolve, reject) => {
                    const transaction = this.db.transaction(['pickups'], 'readonly');
                    const store = transaction.objectStore('pickups');
                    const request = store.getAll();
                    request.onsuccess = (event) => { resolve(event.target.result || []); };
                    request.onerror = () => { reject('Error fetching pickups'); };
                });
            }

            async updatePickup(pickupData) {
                return new Promise((resolve, reject) => {
                    const transaction = this.db.transaction(['pickups'], 'readwrite');
                    const store = transaction.objectStore('pickups');
                    const request = store.put(pickupData);
                    request.onsuccess = () => { resolve(pickupData); };
                    request.onerror = () => { reject('Error updating pickup'); };
                });
            }

            async deletePickup(id) {
                return new Promise((resolve, reject) => {
                    const transaction = this.db.transaction(['pickups'], 'readwrite');
                    const store = transaction.objectStore('pickups');
                    const request = store.delete(id);
                    request.onsuccess = () => { resolve(); };
                    request.onerror = () => { reject('Error deleting pickup'); };
                });
            }

                return new Promise((resolve, reject) => {

                    const request = store.delete(email);
                    request.onsuccess = () => { resolve(); };
                    request.onerror = () => { reject('Error deleting user'); };
                });

            function getPickupItems(pickup) {
                if (pickup?.items?.length) return pickup.items;
                if (pickup?.wasteType) {
                    const kg = parseFloat(String(pickup.quantity || '').replace(/[^\d.]/g, '')) || 0;
                    return [{ type: pickup.wasteType, weight: kg }];
                }
                return [];
            }

            function getPickupSummary(pickup) {
                const items = getPickupItems(pickup);
                if (!items.length) return 'Waste pickup';
                if (items.length === 1) return `${items[0].type} (${items[0].weight} kg)`;
                const total = items.reduce((sum, i) => sum + (parseFloat(i.weight) || 0), 0);
                return `${items.length} items (${total.toFixed(1)} kg)`;
            }

            function getTotalPickupWeight(items) {
                return items.reduce((sum, i) => sum + (parseFloat(i.weight) || 0) * (i.quantity || 1), 0);
            }

            function aggregateWasteFromPickups(pickups) {
                const totals = {};
                let totalKg = 0;
                pickups.forEach(p => {
                    if (p.status !== 'Completed' && p.status !== 'Collected') return;
                    const items = p.items?.length ? p.items : (p.wasteType ? [{ type: p.wasteType, weight: parseFloat(p.quantity) || 0 }] : []);
                    items.forEach(item => {
                        const type = item.type || 'Other';
                        const w = parseFloat(item.weight) || 0;
                        totals[type] = (totals[type] || 0) + w;
                        totalKg += w;
                    });
                });
                const adjust = AdminStore.getWasteAdjust();
                Object.keys(adjust).forEach(k => { totals[k] = (totals[k] || 0) + (parseFloat(adjust[k]) || 0); });
                const finalTotal = Object.values(totals).reduce((s, v) => s + v, 0);
                return { totals, totalKg: finalTotal };
            }

                    document.getElementById('dispatcherUsername').classList.remove('input-invalid');
                    document.getElementById('dispatcherUsername').classList.add('input-valid');
                }
                
                if (password !== 'eco123') {
                    passwordError.textContent = 'Invalid password';
                    document.getElementById('dispatcherPassword').classList.add('input-invalid');
                    isValid = false;
                } else {
                    document.getElementById('dispatcherPassword').classList.remove('input-invalid');
                    document.getElementById('dispatcherPassword').classList.add('input-valid');
                }
                
                if (isValid) {
                    // Successful login
                    dispatcherLoggedIn = true;
                    sessionStorage.setItem('dispatcherLoggedIn', 'true');
                    document.getElementById('dispatcherLoginScreen').style.display = 'none';
                    document.getElementById('dispatcherPanel').style.display = 'block';
                    
                    // Show success notification
                    notify('✅ Dispatcher login successful!');
                    
                    // Load dispatcher data
                    loadDispatcherData();
                } else {
                    notify('❌ Invalid dispatcher credentials');
                }
            }

            window.dispatchApprove = function(pickupId) {
                const all = getAllPickupRequests();
                const p   = all.find(x => x.id === pickupId);
                if (!p) return;
                const collector = COLLECTORS[Math.floor(Math.random() * COLLECTORS.length)];
                p.status      = 'Assigned';
                p.collector   = collector;
                p.assignedAt  = new Date().toISOString();
                saveAllPickupRequests(all);
                notify('✅ Pickup #' + pickupId + ' approved & assigned to ' + collector.name + '!');
                if (dispatcherLoggedIn) {
                    loadDispatcherData();
                }
            };

            window.dispatchReject = function(pickupId) {
                const all = getAllPickupRequests();
                const p   = all.find(x => x.id === pickupId);
                if (!p) return;
                const reason = prompt('Rejection reason (optional):') || '';
                p.status          = 'Rejected';
                p.rejectedAt      = new Date().toISOString();
                p.rejectionReason = reason;
                saveAllPickupRequests(all);
                notify('Pickup #' + pickupId + ' rejected.');
                if (dispatcherLoggedIn) {
                    loadDispatcherData();
                }
            };

            window.dispatchReassign = function(pickupId) {
                const all = getAllPickupRequests();
                const p   = all.find(x => x.id === pickupId);
                if (!p) return;
                const names = COLLECTORS.map((c,i) => (i+1) + '. ' + c.name + ' (' + c.truck + ')').join('\n');
                const choice = prompt('Select new collector:\n' + names + '\n\nEnter number (1-' + COLLECTORS.length + '):');
                const idx = parseInt(choice) - 1;
                if (idx >= 0 && idx < COLLECTORS.length) {
                    p.collector  = COLLECTORS[idx];
                    p.assignedAt = new Date().toISOString();
                    saveAllPickupRequests(all);
                    notify('✅ Pickup #' + pickupId + ' reassigned to ' + COLLECTORS[idx].name + '!');
                    if (dispatcherLoggedIn) {
                        loadDispatcherData();
                    }
                }
            };

            window.dispatchComplete = function(pickupId) {

                    document.getElementById('kpiTotalPoints').innerText = totalPoints;

                    document.getElementById('kpiActiveUsers').innerText = activeUsers;
                    document.getElementById('kpiAvgPickups').innerText = users.length ? (pickups.length / users.length).toFixed(1) : '0';
                    const latest = users.sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt))[0];
                    document.getElementById('kpiLatestSignup').innerText = latest
                        ? new Date(latest.createdAt).toLocaleDateString('en-PH', { month: 'short', day: 'numeric' })
                        : '—';
                    document.getElementById('sideUserCount').innerText = users.length;

                    const payCount = AdminStore.getPaymentConsents().length;
                    const sidePay = document.getElementById('sidePaymentCount');
                    if (sidePay) sidePay.innerText = payCount;
                    document.getElementById('adminTopDate').innerText = new Date().toLocaleDateString('en-PH', { weekday: 'short', month: 'short', day: 'numeric', year: 'numeric' });

                    renderWasteBreakdown(pickups);
                    renderRecentActivity(users, pickups);
                } catch (e) { console.error(e); }

            function renderWasteBreakdown(pickups) {
                const { totals, totalKg } = aggregateWasteFromPickups(pickups);
                const panel = document.getElementById('wasteBreakdownPanel');
                const entries = Object.entries(totals).sort((a, b) => b[1] - a[1]);
                if (!entries.length) {
                    panel.innerHTML = '<p style="color:#aaa;font-size:0.78rem;text-align:center;padding:12px 0;">No completed pickup data yet.</p>';
                    return;
                }
                const colors = ['#2ecc71', '#3498db', '#e74c3c', '#1abc9c', '#9b59b6', '#e67e22', '#f39c12'];
                panel.innerHTML = entries.map(([type, kg], i) => {
                    const pct = totalKg ? Math.round((kg / totalKg) * 100) : 0;
                    return `<div class="waste-bar-item"><div class="waste-bar-label"><span>${esc(getWasteEmoji(type))} ${esc(type)}</span><span>${pct}% · ${kg.toFixed(1)} kg</span></div><div class="waste-bar-track"><div class="waste-bar-fill" style="width:${pct}%;background:${colors[i % colors.length]};"></div></div></div>`;
                }).join('');
            }

            function renderRecentActivity(users, pickups) {
                const panel = document.getElementById('recentActivityPanel');
                const acts = [];
                pickups.slice(0, 4).forEach(p => {
                    acts.push({ icon: 'green', fa: 'fa-truck', text: `Pickup #${p.id} — ${esc(p.residentName)} (${esc(p.status)})`, time: p.dateLabel });
                });
                users.slice(-2).forEach(u => {
                    acts.push({ icon: 'blue', fa: 'fa-user-plus', text: `New resident: ${esc(u.name || u.email)}`, time: new Date(u.createdAt).toLocaleDateString('en-PH') });
                });
                panel.innerHTML = acts.length ? acts.map(a => `
                    <div class="activity-item">
                        <div class="activity-dot ${a.icon}"><i class="fas ${a.fa}"></i></div>
                        <div><div class="activity-text">${a.text}</div><div class="activity-time">${a.time}</div></div>
                    </div>`).join('') : '<p style="color:#aaa;font-size:0.78rem;text-align:center;padding:12px 0;">No recent activity</p>';
            }

            function exportPaymentConsentsCsv() {
                const records = AdminStore.getPaymentConsents();
                if (!records.length) {
                    notify('No payment records to export');
                    return;
                }
                const csvEscape = v => `"${String(v ?? '').replace(/"/g, '""')}"`;
                let csv = 'Record ID,Resident Name,Email,Contact,Payment Mode,Payment Details,Consent Given,Consent Text,Pickup ID,Date\n';
                records.forEach(r => {
                    const details = formatPaymentDetailsLabel(r.paymentMode, r.paymentDetails);
                    csv += [
                        csvEscape(r.id),
                        csvEscape(r.residentName),
                        csvEscape(r.userEmail),
                        csvEscape(r.contactNumber),
                        csvEscape(r.paymentMode),
                        csvEscape(details),
                        r.consentGiven ? 'Yes' : 'No',
                        csvEscape(r.consentText),
                        csvEscape(r.pickupId != null ? r.pickupId : ''),
                        csvEscape(r.createdAt ? new Date(r.createdAt).toLocaleString('en-PH') : '')
                    ].join(',') + '\n';
                });
                const blob = new Blob([csv], { type: 'text/csv' });
                const url = window.URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = `ecosort_payment_logs_${new Date().toISOString().split('T')[0]}.csv`;
                a.click();
                notify('✅ Payment logs CSV exported!');
            }

            async function loadWasteReport() {
                const pickups = await getAdminPickups();
                const { totals, totalKg } = aggregateWasteFromPickups(pickups);
                const panel = document.getElementById('wasteReportPanel');
                const entries = Object.entries(totals).sort((a, b) => b[1] - a[1]);
                const colors = ['#2ecc71', '#3498db', '#e74c3c', '#1abc9c', '#9b59b6', '#e67e22'];

                if (!entries.length) {
                    panel.innerHTML = '<p style="color:#aaa;font-size:0.78rem;text-align:center;padding:12px 0;">Complete pickups to generate reports, or add manual adjustments below.</p>';
                } else {
                    panel.innerHTML = entries.map(([type, kg], i) => {
                        const pct = totalKg ? Math.round((kg / totalKg) * 100) : 0;
                        return `<div class="waste-bar-item"><div class="waste-bar-label"><span>${esc(getWasteEmoji(type))} ${esc(type)}</span><span>${pct}% · ${kg.toFixed(1)} kg</span></div><div class="waste-bar-track"><div class="waste-bar-fill" style="width:${pct}%;background:${colors[i % colors.length]};"></div></div></div>`;
                    }).join('') + `
                    <div style="margin-top:16px;padding-top:14px;border-top:1px solid #e8f0e8;">
                        <p style="font-size:0.78rem;font-weight:700;color:#0b6e2e;margin-bottom:10px;"><i class="fas fa-sliders-h"></i> Manual adjustment (kg per type)</p>
                        <div class="input-group"><label>Waste type name</label><input type="text" id="wasteAdjustType" placeholder="e.g. Biodegradable"></div>
                        <div style="display:flex;gap:8px;"><div class="input-group" style="flex:1;margin-bottom:0;"><label>Add kg</label><input type="number" id="wasteAdjustKg" step="0.1" min="0" placeholder="0"></div>
                        <button class="btn-sm green" id="saveWasteAdjustBtn" type="button" style="margin-top:22px;"><i class="fas fa-plus"></i> Add</button></div>
                    </div>`;
                    document.getElementById('saveWasteAdjustBtn')?.addEventListener('click', () => {
                        const type = document.getElementById('wasteAdjustType').value.trim();
                        const kg = parseFloat(document.getElementById('wasteAdjustKg').value);
                        if (!type || !kg) return notify('Enter waste type and kg');
                        const adj = AdminStore.getWasteAdjust();
                        adj[type] = (adj[type] || 0) + kg;
                        AdminStore.saveWasteAdjust(adj);
                        notify('Waste report updated');
                        loadWasteReport();
                        loadAdminDashboard();
                    });
                }

                let recyclable = 0, bio = 0, hazardous = 0;
                Object.entries(totals).forEach(([type, kg]) => {
                    const t = type.toLowerCase();
                    if (t.includes('recyclable') || t.includes('paper') || t.includes('metal') || t.includes('glass') || t.includes('textile')) recyclable += kg;
                    else if (t.includes('bio')) bio += kg;
                    else if (t.includes('hazardous') || t.includes('medical') || t.includes('electronic') || t.includes('oil')) hazardous += kg;
                });
                document.getElementById('impactRecyclable').innerText = recyclable.toFixed(1) + ' kg';
                document.getElementById('impactBio').innerText = bio.toFixed(1) + ' kg';
                document.getElementById('impactHazardous').innerText = hazardous.toFixed(1) + ' kg';
            }

            // ========== TRACKING ==========
            function startTrackingSimulation() {
                clearIntervals();
                document.getElementById('trackStatusText').innerText = 'Waiting';
                document.getElementById('trackStatusBadge').innerText = 'Searching';
                document.getElementById('trackProgressFill').style.width = '0%';
                document.getElementById('trackerMessage').innerHTML = 'Matching with nearest city waste collector…';

                const dest = activePickup?.location || currentUser?.address || 'Your address';
                renderTrackingMap('searching', { destination: dest });

                let sec = 15;
                document.getElementById('countdownSeconds').innerText = sec;
                countdownInterval = setInterval(() => {
                    sec--;
                    document.getElementById('countdownSeconds').innerText = sec;
                    if (sec <= 0) {
                        clearInterval(countdownInterval);
                        assignCollector();
                    }
                }, 1000);
            }

            async function assignCollector() {
                if (!activePickup || activePickup.status !== 'Pending') return;
                const collector = pinoyCollectors[Math.floor(Math.random() * pinoyCollectors.length)];
                activePickup.status = 'Driver Assigned';
                activePickup.collector = collector;
                if (activePickup.id) {
                    try { await db.updatePickup(activePickup); } catch (e) { console.error(e); }
                }

                document.getElementById('trackStatusText').innerText = 'On the way';
                document.getElementById('trackStatusBadge').innerText = 'En Route';
                document.getElementById('trackProgressFill').style.width = '40%';
                document.getElementById('collectorInfoArea').innerHTML = `
                    <div class="collector-card">
                        <strong>${collector.name}</strong> ${collector.face}<br>
                        <small>${collector.truck} | ${collector.plate}</small><br>
                        <small>📞 ${collector.phone}</small>
                    </div>`;

                renderTrackingMap('enroute', {
                    origin: collector.origin,
                    destination: activePickup.location,
                    collectorName: collector.name,
                    truckLabel: '🚛'
                });

                document.getElementById('trackerMessage').innerHTML =
                    `<i class="fas fa-truck-moving"></i> ${collector.name} is heading to you from ${collector.origin}`;

                setTimeout(async () => {
                    if (activePickup) {
                        activePickup.status = 'Collected';
                        if (activePickup.id) {
                            try { await db.updatePickup(activePickup); } catch (e) { console.error(e); }
                        }
                        document.getElementById('trackProgressFill').style.width = '100%';
                        document.getElementById('trackStatusText').innerText = 'Collected';
                        document.getElementById('trackStatusBadge').innerText = 'Arrived';
                        renderTrackingMap('arrived', {
                            origin: collector.origin,
                            destination: activePickup.location,
                            collectorName: collector.name
                        });
                        document.getElementById('trackerMessage').innerHTML =
                            '✅ Collector arrived! Thank you for proper waste disposal.';
                        setTimeout(() => completePickup(collector), 2000);
                    }
                }, 7000);
            }

            async function completePickup(collector) {
                if (!activePickup) return;
                const items = getPickupItems(activePickup);
                const totalKg = getTotalPickupWeight(items);

                const itemsHtml = items.map(item =>
                    `<div style="margin:6px 0;"><span class="confirm-waste-chip">${getWasteEmoji(item.type)} ${item.type}</span> — <strong>${parseFloat(item.weight).toFixed(1)} kg</strong></div>`
                ).join('');

                document.getElementById('summaryWasteDetails').innerHTML = items.length > 1
                    ? `<strong>${items.length} waste items · ${totalKg.toFixed(1)} kg total</strong>${itemsHtml}`
                    : `${getWasteEmoji(items[0].type)} ${items[0].type} – ${parseFloat(items[0].weight).toFixed(1)} kg`;

                const pointsEarned = await addHistoryAndPoints(items);
                if (activePickup.id) {
                    activePickup.status = 'Completed';
                    try { await db.updatePickup(activePickup); } catch (e) { console.error(e); }
                }
                document.getElementById('impactMessage').innerHTML =
                    `🌱 The City thanks you for keeping our community clean!<br>Collector: ${collector.name}<br>+${pointsEarned} Eco Points earned${items.length > 1 ? ` (${items.length} items)` : ''}.`;
                activePickup = null;
                pickupWasteItems = [];
                updateDashboard();
                showCustomerScreen('completionScreen');
            }
